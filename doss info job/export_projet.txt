dev.py

import os
import json

# Dossier racine contenant tous les sous-dossiers d'images
DOSSIER_IMAGES = "images"

# Extensions d'images acceptées
EXTENSIONS = (".png", ".jpg", ".jpeg", ".gif", ".webp")

def generer_json_par_dossier():
    # Vérifie que le dossier racine existe
    if not os.path.exists(DOSSIER_IMAGES):
        os.makedirs(DOSSIER_IMAGES)
        print(f"Dossier '{DOSSIER_IMAGES}' créé (aucun sous-dossier pour le moment).")
        return

    # Parcourt tous les sous-dossiers
    sous_dossiers = [d for d in os.listdir(DOSSIER_IMAGES)
                     if os.path.isdir(os.path.join(DOSSIER_IMAGES, d))]

    if not sous_dossiers:
        print("Aucun sous-dossier trouvé dans 'images/'.")
        return

    for dossier in sous_dossiers:
        chemin_complet = os.path.join(DOSSIER_IMAGES, dossier)
        fichiers = [
            f for f in os.listdir(chemin_complet)
            if f.lower().endswith(EXTENSIONS)
        ]

        # Nom du fichier JSON basé sur le nom du dossier
        fichier_json = os.path.join(chemin_complet, f"{dossier}.json")

        # Écriture du JSON
        with open(fichier_json, "w", encoding="utf-8") as f:
            json.dump(fichiers, f, indent=2, ensure_ascii=False)

        print(f"✅ {len(fichiers)} image(s) détectée(s) dans '{dossier}/'")
        print(f"   → Fichier JSON généré : {fichier_json}")

if __name__ == "__main__":
    generer_json_par_dossier()

  (function () {

    /* ── 1. FULLSCREEN ── */
    const fv = document.getElementById('fullscreen-view');
    const fi = document.getElementById('fullscreen-img');

    document.addEventListener('click', function (e) {
      const t = e.target;
      if (

        !t.closest('.module') &&

        fi.src = t.src;

      fv.classList.remove('open');
      fi.src = '';
    });

    /* ── 2. NAV ACTIVE ── */
    const sections = document.querySelectorAll('section[id]');
    const navLinks = document.querySelectorAll('nav a[href^="#"]');
    const navObs = new IntersectionObserver(entries => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          navLinks.forEach(l => l.classList.remove('active'));
          const a = document.querySelector('nav a[href="#' + entry.target.id + '"]');
          if (a) a.classList.add('active');
        }
      });
    }, { threshold: 0.25 });
    sections.forEach(s => navObs.observe(s));

    /* ── 3. VADE GALLERY — IntersectionObserver + progressive load ── */
    function buildGallery(images) {
      const gallery = document.getElementById('gallery-vade');
      gallery.innerHTML = ''; // remove placeholder
      if (!images || !images.length) return;

      // Create all cards immediately but load images lazily
      images.forEach(name => {
        const card = document.createElement('div');
        card.className = 'card';

        const wrapper = document.createElement('div');
        wrapper.className = 'card-img-wrapper';

        const img = document.createElement('img');
        img.alt = name.split('.')[0];
        img.dataset.src = 'images/Vade/' + encodeURIComponent(name);
        // Don't set src yet — lazy observer will do it

        img.addEventListener('load', function () {
          img.classList.add('loaded');
          wrapper.classList.add('img-done');
        });
        img.addEventListener('error', function () {
          wrapper.classList.add('img-done'); // remove shimmer even on error
        });

        const caption = document.createElement('div');
        caption.className = 'caption';
        caption.textContent = name.split('.')[0];

        wrapper.appendChild(img);
        card.appendChild(wrapper);
        card.appendChild(caption);
        gallery.appendChild(card);
      });

      // Observe each lazy image
      const imgObs = new IntersectionObserver((entries, obs) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            const el = entry.target;
            if (el.dataset.src) {
              el.src = el.dataset.src;
              delete el.dataset.src;
            }
            obs.unobserve(el);
          }
        });
      }, { rootMargin: '200px' });

      gallery.querySelectorAll('img[data-src]').forEach(img => imgObs.observe(img));
    }

    // Only fetch when gallery section enters viewport
    const gallerySection = document.getElementById('vade');
    let galleryFetched = false;
    const gallerySectionObs = new IntersectionObserver((entries, obs) => {
      entries.forEach(entry => {
        if (entry.isIntersecting && !galleryFetched) {
          galleryFetched = true;
          buildGallery([
            "aware.png","brain wash.png","breathtaking.png","care .png",
            "coffee time.png","flying island croissant (saison 2022).png",
            "hand shake.png","hooked .png","mailbox.png","more space blanc.png",

            "see u soon.png","sleeping turtle.png","summer time.png",

            "vade draw (saison 2022).png","vade logo noir.png","vade space travel (noir).png",
            "vade space travel negatif.png","space traveler front 2.jpg","catalogue.jpg",
            "overflow1.png","coffee time1.png","see you soon1.png",
            "space traveler front 3.jpg","hooked front .jpg"
          ]);
          obs.disconnect();
        }
      });
    }, { rootMargin: '300px' });
    gallerySectionObs.observe(gallerySection);

    /* ── 4. INSTAGRAM — click-to-load ── */
    let igScriptLoaded = false;

    function loadIgScript(cb) {
      if (igScriptLoaded) { cb && cb(); return; }
      const s = document.createElement('script');
      s.async = true;
      s.src = '//www.instagram.com/embed.js';
      s.onload = function () {
        igScriptLoaded = true;
        cb && cb();
      };
      document.body.appendChild(s);
    }

    function injectEmbed(placeholder) {
      const permalink = placeholder.dataset.permalink;
      const container = document.createElement('div');
      container.className = 'insta-embed-container';

      const bq = document.createElement('blockquote');
      bq.className = 'instagram-media';
      bq.dataset.instgrmCaptioned = '';
      bq.dataset.instgrmPermalink = permalink;
      bq.dataset.instgrmVersion = '14';
      bq.style.cssText = 'background:#FFF;border:0;border-radius:3px;box-shadow:0 0 1px 0 rgba(0,0,0,.5),0 1px 10px 0 rgba(0,0,0,.15);padding:0;width:100%;';

      container.appendChild(bq);
      placeholder.replaceWith(container);

      loadIgScript(function () {
        if (window.instgrm && window.instgrm.Embeds) {
          window.instgrm.Embeds.process();
        }
      });
    }

    document.addEventListener('click', function (e) {
      const ph = e.target.closest('.insta-placeholder');
      if (ph) injectEmbed(ph);
    });

    /* ── 5. SKETCHFAB — click-to-load ── */
    document.querySelectorAll('.iframe-lazy-wrapper').forEach(wrapper => {
      const poster = wrapper.querySelector('.iframe-poster');
      if (!poster) return;

        iframe.src = src;
        iframe.style.cssText = 'width:100%;height:100%;border:none;display:block;';
        wrapper.appendChild(iframe);
        wrapper.classList.add('loaded');
      });
    });
